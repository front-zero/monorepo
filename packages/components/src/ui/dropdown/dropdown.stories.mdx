# Overview

Dropdowns are toggleable, contextual overlays for displaying lists of links and more.

<Canvas>
    <Story
        name="Dropdown Controls"
        argTypes={{
            direction: {
                control: {
                    type: `select`,
                    options: ["up", "down", "left", "right"],
                },
            },
            showIcon: {
                control: {
                    type: `inline-radio`,
                    options: [true, false],
                },
            },
            className: {
                control: {
                    type: `text`,
                },
            },
        }}
        args={{
            showIcon: true,
            direction: `down`,
        }}
    >
        {(args) => {
            const { children, ...restArgs } = args;
            return (
                <Dropdown {...restArgs}>
                    <DropdownToggle>Dropdown Toggle</DropdownToggle>
                    <DropdownMenu>
                        <DropdownItem path="#">Action</DropdownItem>
                        <DropdownItem active path="#">
                            Another Action
                        </DropdownItem>
                        <DropdownItem path="#">
                            Something else here
                        </DropdownItem>
                    </DropdownMenu>
                </Dropdown>
            );
        }}
    </Story>
</Canvas>
